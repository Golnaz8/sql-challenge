# Physical

Departments
-
dept_no INT PK
dept_name VARCHAR(255)

Dept_Emp
-
emp_no INT FK >- Employee.emp_no
dept_no INT FK >- Departments.dept_no

Dept_Manager
-
dept_no INT FK >- Departments.dept_no
emp_no INT FK - Employee.emp_no

Employee
-
emp_no INT PK
emp_title_id VARCHAR(10) FK >- Titles.title_id
birth_date DATE
first_name VARCHAR(255)
last_name VARCHAR(255)
sex VARCHAR(10)
hire_date DATE

Salaries
-
emp_no INT FK - Employee.emp_no
salary INT

Titles
-
title_id VARCHAR(10) PK
title VARCHAR(30)
